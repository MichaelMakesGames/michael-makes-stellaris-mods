uncivic_exalted_priesthood_become_follower = {
	if = {
		limit = { uncivic_exalted_priesthood_is_follower = yes }
		uncivic_exalted_priesthood_stop_following = yes
	}
	event_target:uncivic_exalted_priesthood_proposed_leader = {
		prev = {
			set_country_flag = uncivic_exalted_priesthood_follower
			set_country_flag = uncivic_exalted_priesthood_follower_of_@prev
			add_modifier = { modifier = uncivic_exalted_priesthood_follower }
		}
		set_country_flag = uncivic_exalted_priesthood_leader
		set_country_flag = uncivic_exalted_priesthood_leader_of_@prev
		uncivic_exalted_priesthood_recalculate_leader_modifier = yes
		log = "Uncivic Exalted Priesthood: \\[Prev.GetName] started following \\[This.GetName] on \\[GetDate]"
		if = {
			limit = { NOT = { has_country_flag = uncivic_exalted_priesthood_galactic_faith_formed } }
			set_country_flag = uncivic_exalted_priesthood_galactic_faith_formed
			every_country = {
				limit = {
					is_country_type = default
					has_communications = prev
				}
				country_event = { id = uncivic_exalted_priesthood.721 }
			}
		}
	}
}

uncivic_exalted_priesthood_stop_following = {
	if = {
		limit = {
			any_country = { uncivic_exalted_priesthood_is_leader = { OF = prev } }
		}
		random_country = {
			limit = { uncivic_exalted_priesthood_is_leader = { OF = prev } }
			prev = {
				remove_country_flag = uncivic_exalted_priesthood_follower
				remove_country_flag = uncivic_exalted_priesthood_follower_of_@prev
				remove_modifier = uncivic_exalted_priesthood_follower
			}
			remove_country_flag = uncivic_exalted_priesthood_leader_of_@prev
			if = {
				limit = {
					NOT = { any_country = { uncivic_exalted_priesthood_is_follower = { OF = prev } } }
				}
				remove_country_flag = uncivic_exalted_priesthood_leader
			}
			uncivic_exalted_priesthood_recalculate_leader_modifier = yes
			create_message = {
				type = UNCIVIC_EXALTED_PRIESTHOOD_LOST_FOLLOWER
				localization = MESSAGE_UNCIVIC_EXALTED_PRIESTHOOD_LOST_FOLLOWER
				days = 30
				target = PREV
				variable = { type = name localization = COUNTRY scope = PREV }
			}
			log = "Uncivic Exalted Priesthood: \\[Prev.GetName] stopped following \\[This.GetName] on \\[GetDate]"
		}
	} else = {
		remove_country_flag = uncivic_exalted_priesthood_follower
		remove_modifier = uncivic_exalted_priesthood_follower
		log = "Uncivic Exalted Priesthood: \\[Prev.GetName] stopped following non-existent empire on \\[GetDate]"
	}
}
