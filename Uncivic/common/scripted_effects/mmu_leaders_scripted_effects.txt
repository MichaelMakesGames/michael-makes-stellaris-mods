mmu_leaders_add_trait_if_absent = {
	if = {
		limit = { NOT = { has_trait = $TRAIT$ } }
		add_trait = $TRAIT$
	}
}

mmu_leaders_remove_trait_if_present = {
	if = {
		limit = { has_trait = $TRAIT$ }
		remove_trait = $TRAIT$
	}
}
