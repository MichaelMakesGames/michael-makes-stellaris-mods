@level_2_multiplier = 2
@level_3_multiplier = 5
@level_4_multiplier = 10

@pop_happiness_base = -0.05
@pop_growth_reduction_base = 0.10
@pop_assembly_reduction_base = 0.10
@job_produces_base = -0.05
@amenities_base = -0.05

mmgsiege_shortage_biological_pop_happiness = {
	base = 0
	modifier = {
		add = @pop_happiness_base
		mmgsiege_has_resource_shortage = { RESOURCE = food }
	}
	modifier = {
		multiply = @level_2_multiply
		mmgsiege_has_resource_shortage = { RESOURCE = food LEVEL = 2 }
	}
	modifier = {
		multiply = @level_3_multiply
		mmgsiege_has_resource_shortage = { RESOURCE = food LEVEL = 3 }
	}
	modifier = {
		multiply = @level_4_multiply
		mmgsiege_has_resource_shortage = { RESOURCE = food LEVEL = 4 }
	}
	modifier = {
		multiply = 100
		always = $FOR_DISPLAY|yes$
	}
}
